使用iframe，内嵌一个iframe，显示需要打开的网页。

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8">
    <title>Vite App</title>
  </head>
  <body>
    <iframe id="paymentIframe" name="paymentIframe"
          style="width:100%; height:100vh; border:0; display:block;overflow: hidden;"></iframe>
    
    <script>
      document.addEventListener('DOMContentLoaded', function() {
        const url = 'http://txwx.tyjulink.com/ruoyi/dzz/registerPage'; // 需要打开的网页
        const iframe = document.getElementById('paymentIframe');
        
        if (iframe) {
          iframe.src = url;
          console.log("Iframe source set to:", url);
        } else {
          console.error("Iframe element not found.");
        }
      });
    </script>
  </body>
</html>

```
